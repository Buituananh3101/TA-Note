2
Context
Tạo CartContext — bộ nhớ trung tâm quản lý giỏ hàng toàn cục
📄 src/context/CartContext.tsx

Tạo Context chứa cartList và hàm addToCart — dùng được ở bất kỳ component nào trong App. Dữ liệu được đồng bộ với LocalStorage để không mất khi reload trang.

TSX 
import React, { createContext, useState, useEffect, ReactNode } from "react";
import CartItemModel from "../model/CartItemModel";

interface CartContextType {
    cartList: CartItemModel[];
    addToCart: (item: CartItemModel) => void;
}

export const CartContext = createContext<CartContextType | undefined>(undefined);

export const CartProvider: React.FC<{children: ReactNode}> = ({ children }) => {

    // Khởi tạo State từ LocalStorage — tự động phục hồi giỏ hàng phiên trước
    const [cartList, setCartList] = useState<CartItemModel[]>(() => {
        const savedCart = localStorage.getItem('cart');
        return savedCart ? JSON.parse(savedCart) : [];
    });

    // Mỗi khi cartList thay đổi → lưu lại vào LocalStorage ngay lập tức
    useEffect(() => {
        localStorage.setItem('cart', JSON.stringify(cartList));
    }, [cartList]);

    // Logic Thêm Vào Giỏ: cộng dồn nếu đã có, thêm mới nếu chưa có
    const addToCart = (item: CartItemModel) => {
        setCartList((prevCart) => {
            const existingItem = prevCart.find(c => c.sach.maSach === item.sach.maSach);
            if (existingItem) {
                return prevCart.map(c =>
                    c.sach.maSach === item.sach.maSach
                        ? { ...c, soLuong: c.soLuong + item.soLuong }
                        : c
                );
            }
            return [...prevCart, item];
        });
    };

    return (
        <CartContext.Provider value={{ cartList, addToCart }}>
            {children}
        </CartContext.Provider>
    );
};
3
Provider
Bọc ứng dụng với CartProvider trong index.tsx
📄 src/index.tsx

Để addToCart hoạt động ở bất kỳ component nào (từ ChiTietSanPham đến icon giỏ hàng trên Navbar), hãy bọc <CartProvider> bao quanh toàn bộ App:

TSX 
import React from 'react';
import ReactDOM from 'react-dom/client';
import App from './App';
import { CartProvider } from './context/CartContext';

const root = ReactDOM.createRoot(
    document.getElementById('root') as HTMLElement
);

root.render(
    <React.StrictMode>
        <CartProvider>
            <App />
        </CartProvider>
    </React.StrictMode>
);

3 Hàm handleThemVaoGioHang
TypeScript 
const handleThemVaoGioHang = () => {
    if (sach) {
        if (cartContext) {
            // 1. Tạo CartItem từ model đã định nghĩa ở Bước 1
            const cartItem = new CartItemModel(sach, soLuong);

            // 2. Đẩy vào Context — tự động đồng bộ LocalStorage
            cartContext.addToCart(cartItem);

            // 3. Thông báo thành công (nên dùng react-toastify thay alert)
            alert(`Thêm ${soLuong} cuốn '${sach.tenSach}' vào giỏ hàng!`);
        }
    } else {
        alert("Đang tải dữ liệu sách, vui lòng thử lại!");
    }
};
4 Hàm handleMuaNgay — thêm vào giỏ & chuyển ngay sang trang thanh toán
TypeScript 
// Thêm import ở trên cùng: import { useNavigate } from "react-router-dom";
const navigate = useNavigate();

const handleMuaNgay = () => {
    if (sach && cartContext) {
        const cartItem = new CartItemModel(sach, soLuong);
        cartContext.addToCart(cartItem);
        // Chuyển thẳng sang trang giỏ hàng (đừng quên cấu hình route trong App.tsx)
        navigate("/gio-hang");
    }
};
