Struct là gì và tại sao cần nó?

Hiểu vấn đề trước khi học giải pháp — đây là cách tư duy của lập trình viên giỏi.

Giả sử bạn cần lưu thông tin một sinh viên: tên, tuổi, điểm trung bình. Cách đơn giản nhất:

problem.c
char  ten[50];
int   tuoi;
float diem;

Ổn cho 1 sinh viên. Nhưng nếu cần 100 sinh viên?

nightmare.c
char  ten1[50], ten2[50], ten3[50]; // ... đến ten100
int   tuoi1, tuoi2, tuoi3;         // ... đến tuoi100
float diem1, diem2, diem3;         // ... đến diem100
// 300 biến rời rạc — không thể bảo trì được!
⚠ Vấn đề thực
Các biến rời rạc không có "quan hệ" với nhau trong mắt trình biên dịch. Bạn có thể dễ dàng truyền nhầm tuoi2 trong khi muốn dùng tuoi1.

Giải pháp: struct

Struct (structure) cho phép bạn gom nhiều biến liên quan thành một kiểu dữ liệu mới. Trình biên dịch sẽ cấp phát chúng liền kề trong bộ nhớ:

solution.c
struct SinhVien {
    char  ten[50];   // trường (field / member)
    int   tuoi;
    float diem;
};  // ← dấu ; BẮT BUỘC sau dấu }

struct SinhVien sv1, sv2, sv3;  // 3 sinh viên độc lập
struct SinhVien lop[100];         // mảng 100 sinh viên!

Bộ nhớ trông như thế nào?

Các trường được lưu liền kề nhau trong bộ nhớ (có thể có padding giữa các trường):

struct SinhVien sv = {"Le An", 20, 8.5f}
ten [char×50]
"Le An"
offset 0 · 50 bytes
tuoi [int]
20
offset 52 · 4 bytes
diem [float]
8.5
offset 56 · 4 bytes
ℹ Struct vs Array
Array lưu nhiều phần tử cùng kiểu. Struct lưu nhiều phần tử khác kiểu có liên quan với nhau. Đây là sự khác biệt cốt lõi.

easyBài tập — Khai báo struct đầu tiên
Viết struct ToaDo biểu diễn một điểm trong không gian 2D với tọa độ x, y kiểu float. Sau đó khai báo 2 điểm và in khoảng cách giữa chúng.
Yêu cầu
  • Struct có 2 trường: x và y kiểu float
  • Dùng sqrt() từ math.h để tính khoảng cách
  • In ra: Khoang cach: 5.00
main.cTab = 4 spaces
Dùng struct ToaDo { float x; float y; }; — nhớ dấu ; ở cuối!
Khoảng cách AB: sqrt((b.x-a.x)*(b.x-a.x) + (b.y-a.y)*(b.y-a.y)). Compile với gcc main.c -lm (cần -lm để link thư viện math).
đáp án
#include <stdio.h>
#include <math.h>

struct ToaDo {
    float x;
    float y;
};

int main() {
    struct ToaDo a = {0, 0};
    struct ToaDo b = {3, 4};

    float dx = b.x - a.x;
    float dy = b.y - a.y;
    float d  = sqrt(dx*dx + dy*dy);

    printf("Khoang cach: %.2f\n", d);
    return 0;
}
output
Khoang cach: 5.00

Cú pháp đầy đủ và typedef

Nắm vững các cách khai báo — kể cả cái bẫy hay nhầm nhất.

Cách 1 — Struct thông thường

way1.c
struct NhanVien {
    char  ten[50];
    int   tuoi;
    float luong;
};

// Khai báo biến: phải viết từ khóa "struct" mỗi lần
struct NhanVien nv1;
struct NhanVien nv2, nv3;

Cách 2 — typedef: đặt bí danh

typedef tạo một tên mới (alias) cho kiểu dữ liệu. Sau đó bạn không cần viết struct mỗi lần khai báo:

way2.c
typedef struct {          // không cần tên sau "struct"
    char  ten[50];
    int   tuoi;
    float luong;
} NhanVien;               // ← tên alias đặt Ở ĐÂY

// Khai báo biến: ngắn gọn hơn
NhanVien nv1;
NhanVien nv2, nv3;

Cách 3 — typedef với tên struct (dùng khi struct tự tham chiếu)

way3.c
typedef struct Node {    // có tên sau "struct"
    int          data;
    struct Node *next; // ← cần tên struct để tự tham chiếu
} Node;

Node *head = NULL;  // linked list!

Khởi tạo struct — 3 cách

init.c
NhanVien a;                          // cách 1: chưa khởi tạo (rác)
strcpy(a.ten, "Nguyen A"); a.tuoi = 25; a.luong = 10e6;

NhanVien b = {"Tran B", 30, 15e6};    // cách 2: theo thứ tự

NhanVien c = {                         // cách 3: designated (C99)
    .luong = 12e6,   // thứ tự tùy ý!
    .ten   = "Le C",
    .tuoi  = 28
};
⚠ Bẫy phổ biến nhất
Cách 2 (theo thứ tự) rất nguy hiểm: nếu bạn thêm/đổi thứ tự trường trong struct, tất cả code khởi tạo theo thứ tự sẽ sai im lặng — không có lỗi compiler! Cách 3 (designated) an toàn hơn nhiều trong dự án lớn.

Truy cập & gán thành viên

Toán tử . và những điều không ai nói với bạn.

Để đọc hoặc ghi trường của struct, dùng toán tử chấm .:

access.c
NhanVien nv;
strcpy(nv.ten, "Nguyen Van A");  // gán chuỗi
nv.tuoi  = 28;                    // gán số
nv.luong = 15000000.0f;

printf("%s - %d - %.0f\n", nv.ten, nv.tuoi, nv.luong);

Gán struct cho nhau — hoạt động như bạn nghĩ

copy.c
NhanVien nv1 = {"Tran B", 30, 20e6};
NhanVien nv2 = nv1;   // COPY TOÀN BỘ — hợp lệ trong C!

nv2.tuoi = 35;        // nv1.tuoi vẫn là 30
// Khác mảng: arr2 = arr1 bị LỖI, nhưng struct thì ổn!
ℹ Khác với array!
Bạn không thể gán mảng như arr2 = arr1, nhưng có thể gán struct sv2 = sv1. Tuy nhiên nếu struct chứa con trỏ thì chỉ copy địa chỉ (shallow copy) — cẩn thận với trường hợp này!

Chuỗi trong struct — bẫy quan trọng

string_trap.c
NhanVien nv;

// ❌ SAI — không gán chuỗi trực tiếp sau khởi tạo
nv.ten = "Nguyen A";   // LỖI: array không thể gán như vậy

// ✓ ĐÚNG — dùng strcpy
strcpy(nv.ten, "Nguyen A");

// ✓ ĐÚNG — khi khởi tạo trực tiếp thì được
NhanVien nv2 = {"Nguyen A", 25, 10e6};  // OK!

mediumBài tập — Hoán đổi hai struct
Viết hàm swap(SinhVien *a, SinhVien *b) hoán đổi hai sinh viên. Sau đó in mảng trước và sau khi swap phần tử đầu và cuối.
main.c
Dùng biến tạm: SinhVien tmp = *a; *a = *b; *b = tmp; — gán struct hoàn toàn hợp lệ và copy toàn bộ các trường!
đáp án
void swap(SinhVien *a, SinhVien *b) {
    SinhVien tmp = *a;  // copy toàn bộ struct vào tmp
    *a = *b;           // copy b vào vị trí a
    *b = tmp;          // copy tmp (a cũ) vào b
}
// trong main: swap(&ds[0], &ds[2]);

Con trỏ struct và toán tử ->

Nắm chắc phần này là nền tảng để học linked list, tree và mọi cấu trúc dữ liệu động.

Tại sao cần con trỏ struct?

Khi truyền struct vào hàm theo giá trị, C tạo một bản sao hoàn toàn. Với struct lớn (vài KB), điều này rất tốn kém. Con trỏ chỉ truyền địa chỉ (8 bytes trên 64-bit), không quan trọng struct to cỡ nào.

pointer.c
NhanVien  nv  = {"Nam", 28, 15e6};
NhanVien *ptr = &nv;   // ptr trỏ đến nv

// Hai cách truy cập qua con trỏ — TƯƠNG ĐƯƠNG NHAU:
printf("%s\n", (*ptr).ten);  // cách 1: dereference rồi dùng .
printf("%s\n", ptr->ten);    // cách 2: toán tử -> (ngắn hơn, ưu tiên)

// ptr->ten  ≡  (*ptr).ten — hoàn toàn tương đương
⚠ Bẫy ưu tiên toán tử
*ptr.ten là SAI! Vì . có độ ưu tiên cao hơn *, nên đây thực ra là *(ptr.ten) — nghĩa là "lấy địa chỉ từ ptr rồi dereference" — hoàn toàn khác ý định. Luôn dùng (*ptr).ten hoặc tốt hơn là ptr->ten.

malloc — cấp phát động

malloc.c
#include <stdlib.h>

NhanVien *p = (NhanVien*) malloc(sizeof(NhanVien));
if (p == NULL) { /* xử lý lỗi */ return 1; }

strcpy(p->ten, "Thi C");
p->tuoi  = 26;
p->luong = 12e6;

printf("%s %d\n", p->ten, p->tuoi);
free(p);   // BẮT BUỘC — không free = memory leak!
p = NULL;  // good practice: NULL sau free

Struct lồng nhau

Mô hình hóa dữ liệu thực tế phức tạp — và chuỗi truy cập nhiều tầng.

nested.c
typedef struct { int ngay, thang, nam; } Date;

typedef struct {
    char duong[80];
    char thanhPho[40];
} DiaChi;

typedef struct {
    char    ten[50];
    Date    ngaySinh;   // struct trong struct
    DiaChi  diaChi;     // struct trong struct
} KhachHang;

KhachHang kh = {
    "Nguyen Lan",
    {15, 8, 1998},           // khởi tạo Date
    {"123 Le Loi", "HCM"}    // khởi tạo DiaChi
};

// Truy cập chuỗi nhiều tầng
printf("%d/%d/%d\n",
    kh.ngaySinh.ngay,     // struct.struct.field
    kh.ngaySinh.thang,
    kh.ngaySinh.nam);
✓ Quy tắc chuỗi truy cập
Có bao nhiêu tầng lồng thì có bấy nhiêu dấu chấm (hoặc mũi tên nếu qua con trỏ). kh.ngaySinh.nam = "đi vào kh, lấy ngaySinh, lấy nam".

Con trỏ đến struct lồng nhau

nested_ptr.c
KhachHang *p = &kh;

// Con trỏ đến struct ngoài cùng → dùng ->
// Struct bên trong không phải con trỏ → dùng .
printf("%d\n", p->ngaySinh.nam);  // ✓
printf("%s\n", p->diaChi.thanhPho); // ✓

// Nếu diaChi cũng là con trỏ thì: p->diaChi->thanhPho

Struct với Hàm — giá trị vs con trỏ

Bước cuối và quan trọng nhất — khi nào dùng cái nào, và tại sao.

3 cách truyền struct vào hàm

passing.c
// ① Theo giá trị — nhận BẢN SAO, không sửa được gốc
void inThongTin(NhanVien nv) {           // copy ~58 bytes
    printf("%s\n", nv.ten);
    nv.tuoi++;  // không ảnh hưởng bên ngoài
}

// ② Theo con trỏ — nhận ĐỊA CHỈ, có thể sửa gốc
void tangTuoi(NhanVien *nv) {            // chỉ copy 8 bytes
    nv->tuoi++;  // SỬA biến gốc!
}

// ③ Const pointer — nhận địa chỉ, KHÔNG sửa được
void inAn(const NhanVien *nv) {          // best practice cho read-only
    printf("%s %d\n", nv->ten, nv->tuoi);
    // nv->tuoi++ → lỗi compiler: readonly!
}
ℹ Quy tắc thực tế
Luôn dùng const Type* khi hàm chỉ đọc, Type* khi cần sửa. Chỉ dùng pass-by-value cho struct rất nhỏ (2-3 trường nhỏ) mà bạn muốn hàm không ảnh hưởng biến gốc.

Hàm trả về struct

return_struct.c
// Hàm TẠO struct và trả về — hợp lệ!
NhanVien taoNV(const char *ten, int tuoi, float luong) {
    NhanVien nv;
    strncpy(nv.ten, ten, 49);
    nv.ten[49] = '\0';
    nv.tuoi  = tuoi;
    nv.luong = luong;
    return nv;   // trả về bản sao — OK
}

NhanVien nv1 = taoNV("Le Anh", 27, 13e6);
⚠ Không trả về địa chỉ biến local!
return &nv; trong hàm trên là undefined behavior — biến nv bị hủy khi hàm kết thúc! Hoặc trả về struct theo giá trị, hoặc nhận con trỏ đã được cấp phát từ bên ngoài.

hardBài tập cuối — Sắp xếp mảng struct
Viết đầy đủ: hàm nhap() truyền con trỏ để đọc từ bàn phím, hàm in() dùng const pointer để in, hàm sapXep() sắp xếp bubble sort giảm dần theo lương. Dùng dữ liệu mẫu sẵn có.
final.c
Bubble sort: điều kiện hoán đổi là ds[j].luong < ds[j+1].luong. Hoán đổi 2 struct bằng biến tạm: NhanVien tmp = ds[j]; ds[j] = ds[j+1]; ds[j+1] = tmp;
đáp án đầy đủ
void nhap(NhanVien *nv) {
    printf("Ten: "); scanf("%s", nv->ten);
    printf("Tuoi: "); scanf("%d", &nv->tuoi);
    printf("Luong: "); scanf("%f", &nv->luong);
}

void in(const NhanVien *nv) {
    printf("%-10s %3d tuoi  %.0f VND\n",
           nv->ten, nv->tuoi, nv->luong);
}

void sapXep(NhanVien ds[], int n) {
    for (int i=0; i<n-1; i++)
        for (int j=0; j<n-i-1; j++)
            if (ds[j].luong < ds[j+1].luong) {
                NhanVien tmp = ds[j];
                ds[j]   = ds[j+1];
                ds[j+1] = tmp;
            }
}
