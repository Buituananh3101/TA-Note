Struct là gì? Tại sao cần dùng?

Vấn đề với biến rời rạc

Hãy tưởng tượng bạn cần quản lý thông tin của 100 sinh viên, mỗi sinh viên có tên, tuổi, điểm TB. Nếu dùng biến thông thường:

C — Cách tệ ❌
// Rất khó quản lý, dễ nhầm lẫn!
char ten1[50], ten2[50], ten3[50];
int  tuoi1, tuoi2, tuoi3;
float diem1, diem2, diem3;
// ... và tiếp tục đến 100 biến mỗi loại → 300 biến!
Struct (Structure) cho phép nhóm nhiều biến có liên quan thành một kiểu dữ liệu mới. Đây là nền tảng của lập trình hướng cấu trúc trong C.

Struct giải quyết được gì?

  • Nhóm các thuộc tính liên quan vào một kiểu duy nhất
  • Dễ đọc, dễ bảo trì, dễ mở rộng code
  • Truyền dữ liệu phức tạp vào hàm dễ dàng hơn
  • Nền tảng cho Linked List, Stack, Queue, Tree...

Cú pháp khai báo Struct

Cấu trúc tổng quát

C — Cú pháp
struct TenStruct {
    kiểu_dữ_liệu truong1;
    kiểu_dữ_liệu truong2;
    // thêm các trường khác...
};  // chú ý dấu ; ở cuối!
C — Ví dụ sinh viên
#include <stdio.h>
#include <string.h>

// Định nghĩa struct SinhVien
struct SinhVien {
    char  ten[50];   // họ tên
    int   tuoi;      // tuổi
    float diemTB;   // điểm trung bình
    char  mssv[10];  // mã số SV
};

int main() {
    // Khai báo biến kiểu struct
    struct SinhVien sv1;

    // Gán giá trị cho từng trường
    strcpy(sv1.ten, "Nguyen Van An");
    sv1.tuoi   = 20;
    sv1.diemTB = 8.5;
    strcpy(sv1.mssv, "20110001");

    printf("Ten: %s\n", sv1.ten);
    printf("Tuoi: %d\n", sv1.tuoi);
    printf("Diem TB: %.1f\n", sv1.diemTB);
    return 0;
}
Output
Ten: Nguyen Van An
Tuoi: 20
Diem TB: 8.5

Khai báo bằng typedef — Cách viết gọn hơn

Dùng typedef để bỏ từ khóa struct khi khai báo biến:

C — typedef
// Dùng typedef: alias cho kiểu dữ liệu
typedef struct {
    char  ten[50];
    int   tuoi;
    float diemTB;
} SinhVien;

int main() {
    SinhVien sv1;  // Không cần viết "struct" nữa!
    SinhVien sv2, sv3;
    return 0;
}

C — Các cách khởi tạo
typedef struct {
    char  ten[30];
    int   tuoi;
    float luong;
} NhanVien;

int main() {
    // Cách 1: Gán từng trường
    NhanVien nv1;
    strcpy(nv1.ten, "Tran Thi B");
    nv1.tuoi  = 28;
    nv1.luong = 15000000;

    // Cách 2: Khởi tạo trực tiếp (theo thứ tự)
    NhanVien nv2 = {"Le Van C", 32, 20000000};

    // Cách 3: Designated initializer (C99)
    NhanVien nv3 = {
        .ten   = "Pham Thi D",
        .luong = 18000000,
        .tuoi  = 25    // thứ tự không quan trọng
    };

    printf("NV2: %s - %d tuoi - %.0f VND\n",
           nv2.ten, nv2.tuoi, nv2.luong);
    return 0;
}
Output

Struct lồng nhau (Nested Struct)

Struct có thể chứa struct khác làm thành viên — rất hữu ích để mô hình hóa dữ liệu thực tế phức tạp!
C — Ví dụ thực tế
#include <stdio.h>

// Struct con: Ngày tháng
typedef struct {
    int ngay, thang, nam;
} NgayThang;

// Struct con: Địa chỉ
typedef struct {
    char duong[100];
    char quan[50];
    char thanhPho[50];
} DiaChi;

// Struct chính: Khách hàng
typedef struct {
    char      ten[60];
    NgayThang ngaySinh;   // struct lồng struct!
    DiaChi    diaChi;     // struct lồng struct!
    long      soDienThoai;
} KhachHang;

int main() {
    KhachHang kh = {
        .ten = "Nguyen Thi Lan",
        .ngaySinh = {15, 8, 1995},
        .diaChi = {
            "123 Nguyen Hue",
            "Quan 1",
            "Ho Chi Minh"
        },
        .soDienThoai = 0901234567
    };

    printf("Ten: %s\n", kh.ten);
    printf("Sinh: %d/%d/%d\n",
           kh.ngaySinh.ngay,
           kh.ngaySinh.thang,
           kh.ngaySinh.nam);
    printf("Dia chi: %s, %s, %s\n",
           kh.diaChi.duong,
           kh.diaChi.quan,
           kh.diaChi.thanhPho);
    return 0;
}
Output
Ten: Nguyen Thi Lan
Sinh: 15/8/1995
Dia chi: 123 Nguyen Hue, Quan 1, Ho Chi Minh

Con trỏ và Struct

Tại sao cần con trỏ struct?

  • Truyền struct vào hàm mà không sao chép dữ liệu (tiết kiệm bộ nhớ)
  • Thay đổi giá trị struct bên trong hàm (truyền theo địa chỉ)
  • Xây dựng cấu trúc dữ liệu động (linked list, tree...)
C — Con trỏ struct
#include <stdio.h>
#include <stdlib.h>

typedef struct {
    char  ten[30];
    int   tuoi;
    float diem;
} SinhVien;

int main() {
    SinhVien  sv  = {"Le Minh", 21, 9.0};
    SinhVien *ptr = &sv;   // con trỏ trỏ vào sv

    // Hai cách truy cập qua con trỏ (tương đương):
    printf("Cach 1: %s\n", (*ptr).ten);   // dùng (*ptr).field
    printf("Cach 2: %s\n", ptr->ten);     // dùng ptr->field ✓

    // Thay đổi qua con trỏ
    ptr->tuoi = 22;
    printf("Tuoi moi: %d\n", sv.tuoi);  // sv.tuoi cũng thay đổi!

    // Cấp phát động
    SinhVien *sv2 = (SinhVien*)malloc(sizeof(SinhVien));
    strcpy(sv2->ten, "Pham Anh");
    sv2->tuoi = 20;
    sv2->diem = 7.5;
    printf("Dong: %s - %d - %.1f\n",
           sv2->ten, sv2->tuoi, sv2->diem);
    free(sv2);  // giải phóng bộ nhớ
    return 0;
}
Output
Cach 1: Le Minh
Cach 2: Le Minh
Tuoi moi: 22
Dong: Pham Anh - 20 - 7.5
Luôn dùng free() sau malloc() để tránh rò rỉ bộ nhớ (memory leak). Đây là lỗi phổ biến nhất khi làm việc với con trỏ struct!

Mảng các Struct

C — Quản lý danh sách sinh viên
#include <stdio.h>
#include <string.h>

#define N 3

typedef struct {
    char  ten[30];
    float diem;
} SinhVien;

// Hàm in danh sách
void inDanhSach(SinhVien ds[], int n) {
    printf("%-5s %-20s %s\n", "STT", "Ten", "Diem");
    printf("%-5s %-20s %s\n", "---", "---", "----");
    for (int i = 0; i < n; i++) {
        printf("%-5d %-20s %.1f\n",
               i+1, ds[i].ten, ds[i].diem);
    }
}

// Tìm sinh viên điểm cao nhất
SinhVien timDiemCao(SinhVien ds[], int n) {
    int maxIdx = 0;
    for (int i = 1; i < n; i++)
        if (ds[i].diem > ds[maxIdx].diem)
            maxIdx = i;
    return ds[maxIdx];
}

int main() {
    SinhVien ds[N] = {
        {"Nguyen An",   8.5},
        {"Tran Binh",  9.2},
        {"Le Cuong",   7.8}
    };

    inDanhSach(ds, N);

    SinhVien svGioi = timDiemCao(ds, N);
    printf("\nDiem cao nhat: %s (%.1f)\n",
           svGioi.ten, svGioi.diem);
    return 0;
}
Output
STT Ten Diem
--- --- ----
1 Nguyen An 8.5
2 Tran Binh 9.2
3 Le Cuong 7.8

Diem cao nhat: Tran Binh (9.2)

Struct với Hàm — Truyền theo giá trị vs con trỏ

Kiểu truyềnCú phápThay đổi ngoài hàm?Hiệu năng
Theo giá trịvoid f(SinhVien sv)❌ KhôngTốn (sao chép)
Theo con trỏvoid f(SinhVien *sv)✅ CóTốt (chỉ địa chỉ)
Const pointervoid f(const SinhVien *sv)❌ KhôngTốt (an toàn)
C — So sánh hai cách truyền
#include <stdio.h>

typedef struct { char ten[20]; int tuoi; } Nguoi;

// Truyền theo giá trị: hàm nhận BẢN SAO
void tangTuoiSai(Nguoi n) {
    n.tuoi++;  // chỉ thay đổi bản sao!
}

// Truyền theo con trỏ: hàm nhận ĐỊA CHỈ THẬT
void tangTuoiDung(Nguoi *n) {
    n->tuoi++;  // thay đổi biến gốc!
}

// In thông tin (dùng const để bảo vệ dữ liệu)
void inNguoi(const Nguoi *n) {
    printf("%s - %d tuoi\n", n->ten, n->tuoi);
}

int main() {
    Nguoi a = {"Nam", 25};

    tangTuoiSai(a);
    printf("Sau tangTuoiSai: ");
    inNguoi(&a);  // vẫn 25!

    tangTuoiDung(&a);
    printf("Sau tangTuoiDung: ");
    inNguoi(&a);  // đã thành 26
    return 0;
}
Output
Sau tangTuoiSai: Nam - 25 tuoi
Sau tangTuoiDung: Nam - 26 tuoi

Bài 1 — Khai báo và in thông tin sách

Tạo struct Sach gồm: tên sách (chuỗi), tác giả (chuỗi), năm xuất bản (int), giá (float). Khai báo 2 cuốn sách và in thông tin ra màn hình.

code của bạn
Gợi ý: Thành viên gồm char tenSach[100], char tacGia[50], int namXB, float gia. Dùng strcpy() để gán chuỗi.
Đáp án
#include <stdio.h>
#include <string.h>

typedef struct {
    char  tenSach[100];
    char  tacGia[50];
    int   namXB;
    float gia;
} Sach;

int main() {
    Sach s1, s2;

    strcpy(s1.tenSach, "Lap trinh C co ban");
    strcpy(s1.tacGia,  "Nguyen Van A");
    s1.namXB = 2020;
    s1.gia   = 85000;

    Sach s2 = {"Cau truc du lieu", "Tran Thi B", 2019, 120000};

    printf("Ten: %s | Tac gia: %s | Nam: %d | Gia: %.0f\n",
           s1.tenSach, s1.tacGia, s1.namXB, s1.gia);
    printf("Ten: %s | Tac gia: %s | Nam: %d | Gia: %.0f\n",
           s2.tenSach, s2.tacGia, s2.namXB, s2.gia);
    return 0;
}

Bài 2 — Tính diện tích hình chữ nhật

Tạo struct HinhChuNhat gồm chiều dài và chiều rộng (float). Viết hàm tinhDienTich() nhận con trỏ const struct và trả về diện tích.

code của bạn
Prototype hàm: float tinhDienTich(const HinhChuNhat *h) → return h->chieuDai * h->chieuRong
Đáp án
#include <stdio.h>

typedef struct {
    float chieuDai;
    float chieuRong;
} HinhChuNhat;

float tinhDienTich(const HinhChuNhat *h) {
    return h->chieuDai * h->chieuRong;
}

float tinhChuVi(const HinhChuNhat *h) {
    return 2 * (h->chieuDai + h->chieuRong);
}

int main() {
    HinhChuNhat h = {5.0, 3.0};
    printf("Dien tich: %.2f\n", tinhDienTich(&h));
    printf("Chu vi: %.2f\n",   tinhChuVi(&h));
    return 0;
}

Bài 3 — Quản lý danh sách sản phẩm

Tạo struct SanPham gồm: tên (chuỗi), giá (float), số lượng (int). Nhập vào 5 sản phẩm, sau đó: (1) In danh sách, (2) Tính tổng giá trị kho hàng, (3) Tìm sản phẩm đắt nhất.

code của bạn
Tổng giá trị = tổng (gia * soLuong) của từng sản phẩm. Dùng vòng lặp for để duyệt mảng.
Đáp án
#include <stdio.h>
#define N 5

typedef struct {
    char  ten[50];
    float gia;
    int   soLuong;
} SanPham;

void inDanhSach(SanPham ds[], int n) {
    printf("%-15s %12s %8s\n", "Ten", "Gia", "SL");
    for (int i=0; i<n; i++)
        printf("%-15s %12.0f %8d\n",
               ds[i].ten, ds[i].gia, ds[i].soLuong);
}

float tinhTongGiaTri(SanPham ds[], int n) {
    float tong = 0;
    for (int i=0; i<n; i++)
        tong += ds[i].gia * ds[i].soLuong;
    return tong;
}

SanPham timDatNhat(SanPham ds[], int n) {
    int maxI = 0;
    for (int i=1; i<n; i++)
        if (ds[i].gia > ds[maxI].gia) maxI = i;
    return ds[maxI];
}

int main() {
    SanPham ds[N] = {
        {"Laptop", 15000000, 5},
        {"Mouse",  250000,   20},
        {"Keyboard", 500000, 15},
        {"Monitor", 4500000, 8},
        {"Headset", 800000,  12}
    };

    inDanhSach(ds, N);
    printf("\nTong gia tri kho: %.0f VND\n", tinhTongGiaTri(ds, N));

    SanPham dat = timDatNhat(ds, N);
    printf("San pham dat nhat: %s (%.0f VND)\n", dat.ten, dat.gia);
    return 0;
}

Bài 4 — Struct lồng nhau: Hóa đơn

Xây dựng hệ thống hóa đơn: Tạo struct NgayThang (ngay, thang, nam), struct KhachHang (ten, soDT), struct HoaDon (ma, khachHang, ngayMua, tongTien). In ra hóa đơn theo định dạng đẹp.

code của bạn
Truy cập ngày mua: hd.ngayMua.ngay, hd.ngayMua.thang...
Truy cập khách hàng: hd.khachHang.ten...
Đáp án
#include <stdio.h>
#include <string.h>

typedef struct { int ngay, thang, nam; } NgayThang;
typedef struct { char ten[50]; char soDT[15]; } KhachHang;

typedef struct {
    char       maHD[10];
    KhachHang  khachHang;
    NgayThang  ngayMua;
    float      tongTien;
} HoaDon;

void inHoaDon(const HoaDon *hd) {
    printf("================================\n");
    printf("       HOA DON BAN HANG\n");
    printf("================================\n");
    printf("Ma HD    : %s\n", hd->maHD);
    printf("Ngay mua : %02d/%02d/%d\n",
           hd->ngayMua.ngay,
           hd->ngayMua.thang,
           hd->ngayMua.nam);
    printf("Khach    : %s\n", hd->khachHang.ten);
    printf("SoDT     : %s\n", hd->khachHang.soDT);
    printf("--------------------------------\n");
    printf("TONG TIEN: %.0f VND\n", hd->tongTien);
    printf("================================\n");
}

int main() {
    HoaDon hd = {
        "HD001",
        {"Nguyen Van A", "0901234567"},
        {9, 4, 2026},
        1250000
    };
    inHoaDon(&hd);
    return 0;
}

Bài 5 — Sắp xếp mảng struct theo điểm

Cho mảng 5 sinh viên. Viết hàm sắp xếp bằng Bubble Sort theo điểm TB giảm dần. In danh sách trước và sau khi sắp xếp, cùng với xếp loại (Giỏi ≥8.5, Khá ≥7.0, TB ≥5.0, Yếu).

code của bạn
Bubble sort: 2 vòng lặp lồng nhau. Hoán đổi 2 struct: dùng biến tạm SinhVien tmp = ds[j]; ds[j] = ds[j+1]; ds[j+1] = tmp;
Đáp án
#include <stdio.h>
#include <string.h>

typedef struct { char ten[30]; float diemTB; } SinhVien;

const char* xepLoai(float d) {
    if (d >= 8.5) return "Gioi";
    if (d >= 7.0) return "Kha";
    if (d >= 5.0) return "Trung binh";
    return "Yeu";
}

void sapXepGiamDan(SinhVien ds[], int n) {
    for (int i=0; i<n-1; i++)
        for (int j=0; j<n-i-1; j++)
            if (ds[j].diemTB < ds[j+1].diemTB) {
                SinhVien tmp = ds[j];
                ds[j]   = ds[j+1];
                ds[j+1] = tmp;
            }
}

void in(SinhVien ds[], int n) {
    for (int i=0; i<n; i++)
        printf("%d. %-15s %.1f  [%s]\n",
               i+1, ds[i].ten,
               ds[i].diemTB,
               xepLoai(ds[i].diemTB));
}

int main() {
    SinhVien ds[] = {
        {"Nguyen An", 7.5}, {"Tran Binh", 9.1},
        {"Le Cuong",  4.8}, {"Pham Dung", 8.5},
        {"Hoang Em", 6.2}
    };
    int n = 5;
    printf("=== TRUOC KHI SAP XEP ===\n"); in(ds, n);
    sapXepGiamDan(ds, n);
    printf("\n=== SAU KHI SAP XEP ===\n");  in(ds, n);
    return 0;
}
